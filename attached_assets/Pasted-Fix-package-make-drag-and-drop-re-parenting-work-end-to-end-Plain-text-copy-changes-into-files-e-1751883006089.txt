8 Server – rebuild PUT route for parent change

File server/routes/contacts.ts

Locate existing PUT /api/contacts/:id and replace entire handler with:

router.put('/api/contacts/:id', async (req, res, next) => {
  const id = req.params.id;              // UUID string
  const parse = reparentSchema.safeParse(req.body);
  if (!parse.success) return res.status(400).json({ error: 'Bad payload' });

  try {
    const { parentId } = parse.data;     // may be null
    const updated = await db
      .update(contacts)
      .set({ parentId })
      .where(eq(contacts.id, id))
      .returning();

    if (!updated.length) return res.status(404).json({ error: 'Not found' });

    await db.insert(contact_changes).values({
      contact_id: id,
      op: 'update',
      payload: { parentId },
      device_id: req.headers['x-device-id']
    });

    res.json(updated[0]);
  } catch (err) {
    next(err);
  }
});

(Uses simple update; no transaction wrapper.)
